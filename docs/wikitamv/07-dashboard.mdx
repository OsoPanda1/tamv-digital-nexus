# Dashboard de Monitoreo

## Visión General

El Dashboard de Monitoreo de TAMV MD-X4™ proporciona visibilidad en tiempo real del estado del ecosistema, incluyendo métricas de nodos federados, usuarios activos, rendimiento del sistema y seguridad.

## Tabla de Nodos Federados (48 Nodos)

### Estructura de Datos

```typescript
interface FederatedNode {
  id: string;
  name: string;
  region: string;
  country: string;
  status: 'online' | 'warning' | 'offline' | 'maintenance';
  metrics: {
    latency: number;      // ms
    load: number;         // 0-100%
    health: number;       // 0-100
    uptime: number;       // days
  };
  capabilities: {
    quantum: boolean;
    bci: boolean;
    xr: boolean;
    ai: boolean;
  };

}
```

### Componente de Tabla

```typescript
const NodesTable = ({ nodes, membershipTier }: { nodes: FederatedNode[], membershipTier: MembershipTier }) => {
  // Visibilidad según tier
  const visibleNodes = nodes.filter(node => {
    if (membershipTier === 'FREE') return false;
    if (membershipTier === 'STARTER') return nodes.slice(0, 10);
    if (membershipTier === 'PRO') return nodes.slice(0, 25);
    return nodes; // BUSINESS+ ve todos
  });
  
  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead>ID</TableHead>
          <TableHead>Nombre</TableHead>
          <TableHead>Región</TableHead>
          <TableHead>Latencia</TableHead>
          <TableHead>Carga</TableHead>
          <TableHead>Salud</TableHead>
          <TableHead>Estado</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {visibleNodes.map(node => (
          <TableRow key={node.id}>
            <TableCell className="font-mono">{node.id}</TableCell>
            <TableCell>{node.name}</TableCell>
            <TableCell>{node.region}</TableCell>
            <TableCell>{node.metrics.latency}ms</TableCell>
            <TableCell>
              <Progress value={node.metrics.load} />
            </TableCell>
            <TableCell>{node.metrics.health}%</TableCell>
            <TableCell>
              <Badge variant={node.status === 'online' ? 'success' : 'warning'}>
                {node.status}
              </Badge>
            </TableCell>
          </TableRow>
        ))}
      </TableBody>
    </Table>
  );
};
```
